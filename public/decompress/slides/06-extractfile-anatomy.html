const extractFile = (input, output, opts) => runPlugins(input, opts).then(files => {
  // ... strip / filter / map ...

  return Promise.all(files.map(x => {                        //  ◄ ALL entries in parallel
    const dest = path.join(output, x.path);
    const mode = x.mode & ~process.umask();

    if (x.type === 'directory') { /* mkdir + utimes */ }

    return makeDir(output)
      .then(outputPath => fsP.realpath(outputPath))
      .then(realOutputPath => safeMakeDir(path.dirname(dest), realOutputPath)
                              .then(() => realOutputPath))
      .then(realOutputPath => x.type === 'file'
            ? preventWritingThroughSymlink(dest, realOutputPath)            //  check
            : realOutputPath)
      .then(realOutputPath => fsP.realpath(path.dirname(dest)).then(/*containment*/))
      .then(() => {
        if (x.type === 'symlink') return fsP.symlink(x.linkname, dest);     //  use S
        return fsP.writeFile(dest, x.data, {mode});                         //  use F
      });
  }));
});

Red flag #1 Every entry is processed concurrently via Promise.all — no per-path serialisation.

Red flag #2 The symlink check is per-entry, not per-archive.
