fs.writeFile — the dangerous default

node:lib/internal/fs/promises.js (excerpt)
async function writeFile(path, data, options) {
  options = getOptions(options, {
    encoding: 'utf8',
    mode: 0o666,
    flag: 'w',                  //  ◄ O_WRONLY | O_CREAT | O_TRUNC
    flush: false,
  });
  // ...
  const fd = await open(path,
                        options.flag,
                        options.mode);   //  ◄ symlinks are followed
  try {
    await writeFileHandle(fd, data, ...);
  } finally {
    await fd.close();
  }
}
  • ▸
    Default flag is 'w' — no O_NOFOLLOW
  • ▸
    Three syscalls per writeFile: open, write, close — only the first matters for the symlink race
  • ▸
    O_CREAT + symlink-to-nowhere → the file is created at the symlink target; the symlink is no longer dangling
  • ▸
    O_TRUNC → if the target exists, it is truncated. Existing files get clobbered
