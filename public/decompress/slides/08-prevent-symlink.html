const preventWritingThroughSymlink =
  (destination, realOutputPath) => {
    return fsP.readlink(destination)
      .catch(_ => {
        // No file, or not a symlink:
        // not an escape we worry about here.
        return null;
      })
      .then(symlinkPointsTo => {
        if (symlinkPointsTo) {
          throw new Error(
            'Refusing to write into a symlink');
        }
        return realOutputPath;
      });
  };
  • ▸
    Intent — if dest is already a symlink, refuse to write through it
  • ▸
    Mechanism — readlink(). If it succeeds → symlink exists → throw. If ENOENT → safe
  • ▸
    Implicit assumption — the filesystem state seen at check time is the state at use time
