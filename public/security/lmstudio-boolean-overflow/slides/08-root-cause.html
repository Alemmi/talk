src/llama-hparams.cpp
void llama_hparams::set_swa_pattern(
        uint32_t n_pattern, bool dense_first) {
  if (dense_first) {
    for (uint32_t il = 0; il < n_layer; ++il) {     // bound = attacker n_layer
      swa_layers[il] =
        n_pattern == 0 || (il % n_pattern != 0);   // OOB write
    }
  } else {
    for (uint32_t il = 0; il < n_layer; ++il) {     // same primitive (gemma3)
      swa_layers[il] =
        n_pattern == 0 ||
        (il % n_pattern < (n_pattern - 1));
    }
  }
}

The loop writes swa_layers[il] for il ∈ [0, n_layer). The array holds 512 entries; n_layer comes from the file. Any n_layer > 512 writes past the end — up to 2³²−1 iterations, ~16 GB if we wanted.
