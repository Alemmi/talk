case LLM_ARCH_GEMMA3: {
  const bool found_swa = ml.get_key(
    LLM_KV_ATTENTION_SLIDING_WINDOW,
    hparams.n_swa, false);

  if (found_swa && hparams.n_swa > 0) {     // set sliding_window
    hparams.swa_type = LLAMA_SWA_TYPE_STANDARD;
    uint32_t swa_period = 6;
    ml.get_key_or_arr(
      LLM_KV_ATTENTION_SLIDING_WINDOW_PATTERN,
      swa_period, false);
    hparams.set_swa_pattern(swa_period);     // ◄ reaches the vulnerable loop
  } else {
    hparams.swa_type = LLAMA_SWA_TYPE_NONE;
  }
} break;

Why gemma3 It is enabled on the default load path — accepted by LM Studio's loader with no extra opt-in.

Set gemma3.attention.sliding_window > 0 and the branch calls set_swa_pattern. swa_period is also attacker-controlled — it decides the 0/1 pattern that gets written.
